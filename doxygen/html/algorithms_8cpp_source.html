471
472 // Get the initial metrics without heuristics, that will aid is in improving the results
473 vector<int> differences;

475 int numPipes = 0;
476 int maxDifference = 0;
477 Edge<string>* criticalEdge = nullptr;

492 maxDifference = difference;
493 criticalEdge = e;

520map<Edge<string> *, int> get_critical_edges(Graph<string> g, vector<string>& critical_vertexes){
521
522 map<Edge<string> *, int> critical_edges;
523 auto vertexSet = g.getVertexSet();
524
525 int maxCapacity = 0;
526 int difference = 0;
527 int maxDifference = 0;
528 Edge<string>* criticalEdge = nullptr;
529
530 for (auto v : vertexSet) {
531
532 //Verifies if vertex has aready been marked as critical
533 //If it was then the problems have already been solved for this vertex, so we can skip it (in order to avoid infinite loops)
534 int occurences = count(critical_vertexes.begin(), critical_vertexes.end(), v->getInfo());
535
536 if(occurences > 0)
537 continue;
538
539 //Get Vertex Neighbouring Edges
540 int neighbors_count = 0;
541 int neighbors_capacity_sum = 0;
542 double averageCapacity = 0;
543 maxDifference = 0;
544 criticalEdge = nullptr;
545 vector<Edge<string> *> neighbors;
546
547 for (auto e : v->getAdj()) {
548 if(e->getWeight() > 99999) //Ignore edges linking to super sources or super sinks
549 continue;
550 //Get the Edges with the highest capacity and the highest difference between weight and flow
551 //Because if we have a high capacity edge with a high flow input we can't consider it a critical edge
552 //However if we have a high capacity with a high difference (very little flow input) we can consider it a critical edge
553 //And, therefore, we need to update its capacity to a more suitable value
554 difference = e->getWeight() - e->getFlow();
555 if(e->getWeight() > maxCapacity && difference > maxDifference){
556 criticalEdge = e;
557 maxCapacity = e->getWeight();
558 maxDifference = difference;
559 }
560 neighbors_count++;
561 neighbors_capacity_sum += e->getWeight();
562 neighbors.push_back(e);
563 }
564
565 //Get the second biggest capacity neighbor
566 int biggestNeighbor = 0;
567 for(auto e: neighbors){
568 if(e == criticalEdge)
569 continue;
570 else{
571 if(e->getWeight() > biggestNeighbor)
572 biggestNeighbor = e->getWeight();
573 }
574 }
575
576 //The vertex has more than one neighboring edges and its value is above the average
577 if(neighbors_count > 1){
578 averageCapacity = neighbors_capacity_sum / neighbors_count;
579 if(static_cast<double>(maxCapacity) > averageCapacity && criticalEdge != nullptr){
580 critical_vertexes.push_back(v->getInfo());
581 critical_edges[criticalEdge] = biggestNeighbor;
582 }
583 }
584 }
585
586 return critical_edges;
587}

620 //Without using heuristics
621 edmonds_karp(graph, "source", sink);

623
624 //gets the critical edges, as well as the critical vertexes of the graph after running Edmonds Karp
625 map<Edge <string> *, int> critical_edges;
626 vector<string> critical_vertexes;
627 critical_edges = get_critical_edges(graph, critical_vertexes);
628
629 while(true){
630
631 iter_num++;
632 int start_critical_vertexes = critical_vertexes.size();
633
634 //Resets the graph by reseting the flow to its original state (0)
635 for(auto v : graph.getVertexSet()){
636 for(auto e : v->getAdj()){
637 e->setFlow(0);
638 }
639 }
640
641
642 //Find the critical edges and update the graph with the critical edges
643 //In order to find it we have to compare the edge start and finish point
644 for (auto it = critical_edges.begin(); it != critical_edges.end(); ++it) {
645 //Iterate Through All the edges in the graph
646 for(auto v : graph.getVertexSet()){
647 for(auto e : v->getAdj()){
648 if(e->getOrig()->getInfo() == it->first->getOrig()->getInfo() && e->getDest()->getInfo() == it->first->getDest()->getInfo()){
649 e->setWeight(it->second);
650 }
651 }
652 }
653 }
654
655 //Runs Edmonds Karp again and computes the metrics on the graph with the updated critical edges
656 edmonds_karp(graph, "source", sink);
657 double avg_diff = compute_metrics(graph, iter_num);
658
659 if(avg_diff < best_avg){
660 best_graph = graph;
661 best_avg = avg_diff;
662 best_iter_num = iter_num;
663 }
664
665 //Gets the critical vertexes of the graph with the we have just updated critical edges and performed edmonds karp in
666 critical_edges = get_critical_edges(graph, critical_vertexes);
667
668 int end_critical_vertexes = critical_vertexes.size();
669
670 std::cout << "Critical Vertexes start: " << start_critical_vertexes << " Critical Vertexes end: " << end_critical_vertexes << std::endl;
671
672 //The critical vertexes vector didn't update which means that we have found all the critical vertexes
673 if(start_critical_vertexes == end_critical_vertexes)

685 //OBJECTIVE: MINIMIZE THE FLOW VS CAPACITY ON EACH EDGE
686 //Smaller Capacity Edges bottleneck the flow of the path
687 //Therefore, edges with higher capacity suffering from neighboring edges with lower capacity will have
688 //significant capacity-flow >>>> 0 distribution
689 //What can we do:
690 //Locate the neighboring edges(Edges that are connected to the same vertex) and compute their difference
691 //After that we update the edge with the most critical (flow vs capacity) difference with the capacity of the edge with the second highest difference
692 //By doing this we make sure we are not losing a lot of flow, and we are able to balance out the neighborhood of a node
693 //Then we mark the edge as critical and we add it to the critical edges vector
694 //After that we update the visited critical nodes vector with the node that is connected to the critical edge, so
695 //We do not enter in a loop where we try to balance out the same vertex neighborhood multiple times
696 //After that we update the critical edge's capacity and run the algorithm again in order to balance out the graph
697 //We repeat this process until the critical vertexes vector is "full" or there are no more critical edges to be found
698 //Also, this may introduce some stochascity in the results (since we may get different results each time we run the algorithm by modifying
699 //new edges and running Edmonds Karp again (which outputs new results))
700 //So we need to keep track of our best result found so far, so we choose that one instead of the one in the last iteration
